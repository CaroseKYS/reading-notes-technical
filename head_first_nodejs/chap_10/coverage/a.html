JSON parse

传入正确的数据, 不应当抛出异常.
parser.parseAsync(JSON.stringify({name: 'ttt'}), function(err, data){
  assert.equal(null);
  done();
});
传入错误数据, 应当抛出除异常。
parser.parseAsync('{"name": "jack"}}', function(err, data){
  assert.ifError(data);
  assert.notEqual(err, null);
  done();
});
